Advanced Configuration

from aegis import AegisShield, ProtectionConfig

config = ProtectionConfig(
    methods=["redaction", "tokenization"],
    preserve_format=True,
    compliance_mode="HIPAA"
)

client = AegisShield(
    api_key="sk_live_YOUR_KEY",
    config=config
)

# Batch protection for multiple messages
messages = [
    "Patient John Doe, DOB 01/01/1990",
    "Credit card ending in 1234"
]

results = client.protect_batch(messages)
for result in results:
    print(result.protected_text)

Express.js Middleware

const express = require('express');
const { aegisMiddleware } = require('aegis-shield');

const app = express();

// Auto-protect all request bodies
app.use(aegisMiddleware({
  apiKey: 'sk_live_YOUR_KEY',
  fields: ['message', 'userInput'],
  onPIIDetected: (req, piiInfo) => {
    console.log(`PII detected: ${piiInfo.count} items`);
  }
}));

app.post('/chat', (req, res) => {

Spring Boot Integration

@Configuration
public class AegisConfig {

    @Bean
    public AegisShield aegisShield() {
        return new AegisShield.Builder()
            .apiKey(${aegis.api.key})
            .timeout(Duration.ofSeconds(30))
            .retryPolicy(RetryPolicy.exponentialBackoff())
            .build();
    }
}

@Service
public class ChatService {

    @Autowired
    private AegisShield aegis;

    public String processChatMessage(String userInput) {
        ProtectionResult result = aegis.protect(userInput);
        return sendToLLM(result.getProtectedText());
    }
}

Go SDK

High-performance Go SDK designed for microservices and cloud-native applications.

‚úì Context Support ‚úì Goroutine Safe ‚úì gRPC Integration ‚úì Prometheus Metrics

Basic Usage

package main

import (
    "context"
    "fmt"
    "github.com/aegis-shield/go-sdk"
)

func main() {
    client := aegis.NewClient("sk_live_YOUR_KEY")

    result, err := client.Protect(context.Background(), &aegis.ProtectRequest{
        Text: "My phone is (555) 123-4567",
        DetectionLevel: aegis.DetectionLevelStrict,
    })

    if err != nil {
        panic(err)
    }

    fmt.Println(result.ProtectedText) // "My phone is [PHONE_REDACTED]"
}

Microservice Integration

func NewChatHandler(aegisClient *aegis.Client) *ChatHandler {
    return &ChatHandler{
        aegis: aegisClient,
    }
}

func (h *ChatHandler) ProcessMessage(ctx context.Context, req *pb.ChatRequest) (*pb.ChatResponse, error) {
    // Protect user input before sending to LLM
    result, err := h.aegis.Protect(ctx, &aegis.ProtectRequest{
        Text: req.Message,
        PreserveFormat: true,
    })

    if err != nil {
        return nil, fmt.Errorf("protection failed: %w", err)
    }

    // Forward to LLM service
    llmResponse := h.sendToLLM(result.ProtectedText)

    return &pb.ChatResponse{
        Message: llmResponse,
        PiiDetected: result.PiiCount,
    }, nil
}
